Object-Oriented Programming

Discover how to create and use classes and objects in Python, including inheritance and polymorphism.

Example: OOP 
# Class definition
class Person:
    # Constructor
    def __init__(self, name, age):
        self.name = name
        self.age = age
    
    # Method
    def greet(self):
        return f"Hi, I'm {self.name} and I'm {self.age} years old."

# Inheritance
class Student(Person):
    def __init__(self, name, age, student_id):
        super().__init__(name, age)
        self.student_id = student_id
    
    def study(self):
        print(f"{self.name} is studying.")

# Creating objects
person1 = Person("Alice", 25)
student1 = Student("Bob", 20, "S12345")

print(person1.greet())  # Output: Hi, I'm Alice and I'm 25 years old.
student1.study()        # Output: Bob is studying.

Error Handling

Learn how to handle exceptions and errors gracefully in your Python programs.

Example: Error Handling 
# Basic try-except block
try:
    result = 10 / 0
except ZeroDivisionError:
    print("Cannot divide by zero!")

# Handling multiple exceptions
try:
    num = int("abc")
    result = 10 / num
except ValueError:
    print("Invalid number!")
except ZeroDivisionError:
    print("Cannot divide by zero!")
except Exception as e:
    print(f"An error occurred: {e}")
finally:
    print("This always executes")

# Raising custom exceptions
def validate_age(age):
    if age < 0:
        raise ValueError("Age cannot be negative")
    return age

try:
    validate_age(-5)
except ValueError as e:
    print(e)  # Output: Age cannot be negative

File Handling

Learn how to read from and write to files in Python, including working with different file formats.

Example: File Handling 
# Writing to a file
with open("example.txt", "w") as file:
    file.write("Hello, World!\n")
    file.write("This is a second line.\n")

# Reading from a file
with open("example.txt", "r") as file:
    content = file.read()
    print(content)

# Reading line by line
with open("example.txt", "r") as file:
    for line in file:
        print(line.strip())

# Working with JSON
import json

# Writing JSON
data = {"name": "Alice", "age": 25, "courses": ["Python", "Data Science"]}
with open("data.json", "w") as file:
    json.dump(data, file)

# Reading JSON
with open("data.json", "r") as file:
    loaded_data = json.load(file)
    print(loaded_data)
